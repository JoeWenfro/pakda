function pakda_normalizar_mac($mac_raw) {
    $mac = strtoupper(trim((string)$mac_raw));
    $mac = str_replace('-', ':', $mac);
    $mac = preg_replace('/[^0-9A-F:]/', '', $mac);
    if (preg_match('/^[0-9A-F]{12}$/', $mac)) {
        $mac = implode(':', str_split($mac, 2));
    }
    return preg_match('/^([0-9A-F]{2}:){5}[0-9A-F]{2}$/', $mac) ? $mac : '';
}

/**
 * INYECTAR CAMPO HIDDEN CON JAVASCRIPT
 * NO guardamos en cookies/sesión del servidor
 * SOLO en localStorage del navegador
 */
add_action('wp_footer', function() {
    if (is_admin() || wp_doing_ajax()) return;
    
    // Solo en páginas públicas
    ?>
    <script>
    (function() {
        // Capturar MAC de la URL
        var urlParams = new URLSearchParams(window.location.search);
        var mac = urlParams.get('mac');
        
        if (mac) {
            // Limpiar formato
            mac = mac.toUpperCase().replace(/-/g, ':');
            
            // Guardar SOLO en localStorage del navegador
            if (localStorage) {
                var oldMac = localStorage.getItem('pakda_current_mac');
                
                // Si es DIFERENTE, limpiar todo
                if (oldMac && oldMac !== mac) {
                    console.log('PAKDA: Nueva MAC detectada, limpiando cache');
                    localStorage.clear();
                }
                
                localStorage.setItem('pakda_current_mac', mac);
                console.log('PAKDA: MAC guardada en localStorage:', mac);
            }
        }
        
        // En checkout, inyectar campo hidden
        if (document.body.classList.contains('woocommerce-checkout')) {
            var storedMac = localStorage ? localStorage.getItem('pakda_current_mac') : null;
            
            if (storedMac) {
                // Buscar formulario de checkout
                var checkoutForm = document.querySelector('form.checkout');
                
                if (checkoutForm) {
                    // Eliminar campo anterior si existe
                    var oldField = document.getElementById('pakda_mac_field');
                    if (oldField) oldField.remove();
                    
                    // Crear nuevo campo hidden
                    var hiddenField = document.createElement('input');
                    hiddenField.type = 'hidden';
                    hiddenField.id = 'pakda_mac_field';
                    hiddenField.name = 'pakda_mac_from_browser';
                    hiddenField.value = storedMac;
                    
                    checkoutForm.appendChild(hiddenField);
                    console.log('PAKDA: Campo hidden inyectado con MAC:', storedMac);
                }
            } else {
                console.warn('PAKDA: No hay MAC en localStorage');
            }
        }
    })();
    </script>
    <?php
}, 999);

function pakda_activar_wifi_radius($order_id) {
    pakda_log("═══════════════════════════════════════");
    pakda_log("🔔 ACTIVACIÓN - Pedido #$order_id");
    
    // SIN FRENO DE MANO - Permitir reintentos si falla webhook
    // No verificamos _pakda_wifi_activado para permitir que thank you page reintente

    $order = wc_get_order($order_id);
    if (!$order) {
        pakda_log("❌ Pedido no existe");
        pakda_log("═══════════════════════════════════════");
        return;
    }

    $mac = pakda_normalizar_mac($order->get_meta('_customer_mac'));
    
    pakda_log("MAC: " . ($mac ? $mac : "VACÍA"));
    
    if (!$mac) {
        pakda_log("❌ MAC faltante");
        $order->add_order_note('⚠️ PAKDA: MAC faltante');
        pakda_log("═══════════════════════════════════════");
        return;
    }

    // Determinar plan
    $seconds = 7200;
    foreach ($order->get_items() as $item) {
        $n = mb_strtolower($item->get_name());
        
        if (strpos($n, 'ilimitado') !== false) {
            $seconds = 259200;
            pakda_log("Plan: ILIMITADO (3 días)");
        } elseif (strpos($n, '2 horas') !== false || strpos($n, '2 hora') !== false) {
            $seconds = 7200;
            pakda_log("Plan: 2 HORAS");
        }
        break;
    }

    // Conexión DB
    pakda_log("Conectando MySQL...");
    $db = mysqli_init();
    mysqli_options($db, MYSQLI_OPT_CONNECT_TIMEOUT, 5);
    
    if (!@mysqli_real_connect($db, PAKDA_DB_HOST, PAKDA_DB_USER, PAKDA_DB_PASS, PAKDA_DB_NAME, PAKDA_DB_PORT)) {
        pakda_log("❌ Conexión falló: " . mysqli_connect_error());
        pakda_log("═══════════════════════════════════════");
        return;
    }
    
    pakda_log("✅ MySQL OK");
    $db->set_charset('utf8mb4');

    // REPLACE INTO (atómico) en vez de DELETE + INSERT
    pakda_log("Usando REPLACE INTO (atómico)...");
    
    $stmt = $db->prepare("REPLACE INTO radcheck (username, attribute, op, value) VALUES (?, 'Cleartext-Password', ':=', ?)");
    $stmt->bind_param("ss", $mac, $mac);
    $ok1 = $stmt->execute();
    pakda_log("REPLACE radcheck: " . ($ok1 ? "OK" : "FALLO - " . $db->error));
    $stmt->close();
    
    $stmt = $db->prepare("REPLACE INTO radreply (username, attribute, op, value) VALUES (?, 'Mikrotik-Group', ':=', 'full')");
    $stmt->bind_param("s", $mac);
    $ok2 = $stmt->execute();
    pakda_log("REPLACE Mikrotik-Group: " . ($ok2 ? "OK" : "FALLO - " . $db->error));
    $stmt->close();
    
    $time_str = (string)$seconds;
    $stmt = $db->prepare("REPLACE INTO radreply (username, attribute, op, value) VALUES (?, 'Session-Timeout', ':=', ?)");
    $stmt->bind_param("ss", $mac, $time_str);
    $ok3 = $stmt->execute();
    pakda_log("REPLACE Session-Timeout: " . ($ok3 ? "OK" : "FALLO - " . $db->error));
    $stmt->close();
    
    $stmt = $db->prepare("REPLACE INTO radreply (username, attribute, op, value) VALUES (?, 'Idle-Timeout', ':=', '1800')");
    $stmt->bind_param("s", $mac);
    $ok4 = $stmt->execute();
    pakda_log("REPLACE Idle-Timeout: " . ($ok4 ? "OK" : "FALLO - " . $db->error));
    $stmt->close();
    
    $db->close();

    if ($ok1 && $ok2 && $ok3 && $ok4) {
        update_post_meta($order_id, '_pakda_wifi_activado', 'yes');
        update_post_meta($order_id, '_wifi_username', $mac);
        update_post_meta($order_id, '_wifi_time', $seconds);
        
        $order->add_order_note("✅ PAKDA: WiFi activado para $mac");
        pakda_log("🚀 ✅ ÉXITO - WiFi activado");
        
        // ═══════════════════════════════════════════════════════
        // API KICK (PATEO) - Reconectar usuario inmediatamente
        // ═══════════════════════════════════════════════════════
        pakda_log("🦶 Ejecutando API Kick para reconexión inmediata...");
        
        $kick_url = "http://207.180.206.65/api_pakda.php?mac=" . urlencode($mac);
        
        $response = wp_remote_get($kick_url, array(
            'timeout' => 5,
            'sslverify' => false
        ));
        
        if (is_wp_error($response)) {
            pakda_log("⚠️ API Kick falló: " . $response->get_error_message());
        } else {
            $body = wp_remote_retrieve_body($response);
            pakda_log("✅ API Kick ejecutado - Respuesta: " . $body);
        }
        
    } else {
        pakda_log("❌ Falló uno o más REPLACEs");
        $order->add_order_note("❌ PAKDA: Error activación");
    }
    
    pakda_log("═══════════════════════════════════════");
}

function pakda_webhook_authenticate($request) {
    $auth_header = $request->get_header('authorization');
    
    if (empty($auth_header)) {
        pakda_log("⚠️ Webhook sin auth");
        return true;
    }
    
    if (preg_match('/Basic\s+(.*)$/i', $auth_header, $matches)) {
        $credentials = base64_decode($matches[1]);
        list($user, $pass) = explode(':', $credentials, 2);
        
        if ($user === 'pakda_webhook_user' && $pass === 'PakdaW3bh00k2026!') {
            pakda_log("✅ Webhook autenticado");
            return true;
        } else {
            pakda_log("❌ Credenciales incorrectas");
            return new WP_Error('unauthorized', 'Credenciales inválidas', ['status' => 401]);
        }
    }
    
    return true;
}

function pakda_webhook_handler($request) {
    pakda_log("🔔 WEBHOOK Culqi");
    
    $data = $request->get_json_params();
    $oid = $data['data']['metadata']['order_id'] ?? 0;

    if (!$oid) {
        pakda_log("⚠️ Sin order_id");
        return new WP_REST_Response('Missing order_id', 400);
    }

    $order = wc_get_order($oid);
    
    if (!$order) {
        pakda_log("❌ Orden no encontrada");
        return new WP_REST_Response('Order not found', 404);
    }
    
    if ($order->is_paid()) {
        pakda_log("ℹ️ Ya pagada");
        return new WP_REST_Response('Already processed', 200);
    }
    
    $order->payment_complete();
    pakda_log("✅ Marcada como pagada");
    
    return new WP_REST_Response('OK', 200);
}
