function pakda_log($msg) {
    error_log('[PAKDA] ' . $msg);
}

function pakda_normalizar_mac($mac_raw) {
    $mac = strtoupper(trim((string)$mac_raw));
    $mac = str_replace('-', ':', $mac);
    $mac = preg_replace('/[^0-9A-F:]/', '', $mac);
    if (preg_match('/^[0-9A-F]{12}$/', $mac)) {
        $mac = implode(':', str_split($mac, 2));
    }
    return preg_match('/^([0-9A-F]{2}:){5}[0-9A-F]{2}$/', $mac) ? $mac : '';
}

    $mac = $_SESSION['pakda_mac_session'] ?? $_COOKIE['pakda_mac_cookie'] ?? $_GET['mac'] ?? '';

    // Si no tenemos MAC y no estamos en la URL de autologin, redirigimos
    if (empty($mac) && !isset($_GET['pakda_action'])) {
        $gw = 'http://192.168.100.1/login'; 
        $url_actual = home_url($_SERVER['REQUEST_URI']);
        
        wp_redirect($gw . "?dst=" . urlencode($url_actual));
        exit;
    }
});

function pakda_activar_wifi_radius($order_id) {
    // 1. Evitar duplicidad
    if (get_post_meta($order_id, '_pakda_wifi_activado', true)) {
        return;
    }

    $order = wc_get_order($order_id);
    if (!$order) return;

    // 2. Obtener MAC
    $mac = pakda_normalizar_mac($order->get_meta('_customer_mac'));
    
    if (!$mac) {
        pakda_log("❌ Error: Pedido #$order_id no tiene MAC.");
        $order->add_order_note('⚠️ PAKDA: No se pudo activar WiFi (MAC faltante)');
        return;
    }

    // 3. Determinar tiempo
    $seconds = 3600; 
    $plan_name = 'Plan 1 Hora';
    
    foreach ($order->get_items() as $item) {
        $n = mb_strtolower($item->get_name());
        $plan_name = $item->get_name();
        
        if (strpos($n, 'ilimitado') !== false || strpos($n, '5 días') !== false) $seconds = 432000;
        elseif (strpos($n, '2 horas') !== false) $seconds = 7200;
        elseif (strpos($n, '24 horas') !== false || strpos($n, '1 día') !== false) $seconds = 86400;
        elseif (strpos($n, '7 días') !== false) $seconds = 604800;
        elseif (strpos($n, '30 días') !== false) $seconds = 2592000;
        break; 
    }

    // 4. Conexión DB
    $db = mysqli_init();
    mysqli_options($db, MYSQLI_OPT_CONNECT_TIMEOUT, 5);
    
    if (!@mysqli_real_connect($db, PAKDA_DB_HOST, PAKDA_DB_USER, PAKDA_DB_PASS, PAKDA_DB_NAME, PAKDA_DB_PORT)) {
        pakda_log("❌ Error conexión RADIUS: " . mysqli_connect_error());
        return;
    }
    
    $db->set_charset('utf8mb4');

    // 5. Limpieza
    $stmt = $db->prepare("DELETE FROM radcheck WHERE username = ?");
    $stmt->bind_param("s", $mac);
    $stmt->execute(); $stmt->close();
    
    $stmt = $db->prepare("DELETE FROM radreply WHERE username = ?");
    $stmt->bind_param("s", $mac);
    $stmt->execute(); $stmt->close();

    // 6. Inserción
    $stmt = $db->prepare("INSERT INTO radcheck (username, attribute, op, value) VALUES (?, 'Cleartext-Password', ':=', ?)");
    $stmt->bind_param("ss", $mac, $mac);
    $ok1 = $stmt->execute(); $stmt->close();
    
    $stmt = $db->prepare("INSERT INTO radreply (username, attribute, op, value) VALUES (?, 'Mikrotik-Group', ':=', 'full')");
    $stmt->bind_param("s", $mac);
    $ok2 = $stmt->execute(); $stmt->close();
    
    $time_str = (string)$seconds;
    $stmt = $db->prepare("INSERT INTO radreply (username, attribute, op, value) VALUES (?, 'Session-Timeout', ':=', ?)");
    $stmt->bind_param("ss", $mac, $time_str);
    $ok3 = $stmt->execute(); $stmt->close();
    
    $stmt = $db->prepare("INSERT INTO radreply (username, attribute, op, value) VALUES (?, 'Idle-Timeout', ':=', '1800')");
    $stmt->bind_param("s", $mac);
    $ok4 = $stmt->execute(); $stmt->close();
    
    // --- DINÁMICO: IP para Pateo ---
    $query_ip = $db->prepare("SELECT framedipaddress FROM radacct WHERE username = ? ORDER BY acctstarttime DESC LIMIT 1");
    $query_ip->bind_param("s", $mac);
    $query_ip->execute();
    $result_ip = $query_ip->get_result();
    $row_ip = $result_ip->fetch_assoc();
    
    // Si no está en DB, usamos la IP actual del cliente (Salvavidas)
    $user_internal_ip = $row_ip['framedipaddress'] ?? $_SERVER['REMOTE_ADDR'];
    $query_ip->close(); 

    // --- VPN API ---
    $vps_url = "http://207.180.206.65/api_pakda.php"; 
    $api_key = "PAKDA_SECRET_2026"; 

    // Limpieza de IP por seguridad
    $user_internal_ip = filter_var($user_internal_ip, FILTER_VALIDATE_IP) ?: '';

    $url_final = add_query_arg([
        'key' => $api_key,
        'mac' => $mac,
        'uip' => $user_internal_ip
    ], $vps_url);

    wp_remote_get($url_final, [
        'timeout'   => 5,
        'blocking'  => false 
    ]);
    
    $db->close();

    // 7. Final
    if ($ok1 && $ok2 && $ok3 && $ok4) {
        update_post_meta($order_id, '_pakda_wifi_activado', 'yes');
        update_post_meta($order_id, '_wifi_username', $mac);
        update_post_meta($order_id, '_wifi_time', $seconds);
        
        $order->add_order_note("✅ PAKDA: Activado OK para $mac");
        pakda_log("🚀 WiFi activado: MAC $mac | Pedido #$order_id");
    } else {
        pakda_log("❌ Error crítico: Fallo inserción RADIUS");
    }
}

/**
 * 7) WEBHOOK CULQI
 */
add_action('rest_api_init', function () {
    register_rest_route('pakda/v1', '/webhook-culqi', [
        'methods'  => 'POST',
        'callback' => 'pakda_webhook_handler',
        'permission_callback' => '__return_true'
    ]);
});

function pakda_webhook_handler($request) {
    $data = $request->get_json_params();
    $oid = $data['data']['metadata']['order_id'] ?? 0;

    if ($oid) {
        $order = wc_get_order($oid);
        if ($order && !$order->is_paid()) {
            $order->payment_complete();
            pakda_log("✅ Webhook Culqi: Orden #$oid pagada");
            return new WP_REST_Response('OK', 200);
        }
    }
    return new WP_REST_Response('Processed', 200);
}

add_action('woocommerce_thankyou', function($order_id) {
    if (!$order_id) return;
    
    // 1. Intentamos recuperar el token
    $token = get_post_meta($order_id, '_pakda_autologin_token', true);
    
    // 2. Si no existe o está vacío, lo forzamos con wp_generate_password (Más seguro que bin2hex)
    if (empty($token)) {
        $token = wp_generate_password(32, false); 
        update_post_meta($order_id, '_pakda_autologin_token', $token);
    }
    
    // 3. Construimos la URL
    $url = add_query_arg([
        'pakda_action' => 'autologin', 
        'token' => $token
    ], home_url('/'));
    
    // 4. Debug: Guardamos en el log qué URL se generó
    pakda_log("Redirección creada para Orden #$order_id. Token: $token");
    
    // 5. Mostramos Botón Manual + Redirección JS
    if (!empty($token)) {
        echo '<div style="text-align:center;margin:20px;padding:30px;border:2px solid #34d399;background:#f0fff4;border-radius:15px;box-shadow: 0 4px 6px rgba(0,0,0,0.1);">
